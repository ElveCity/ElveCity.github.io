# OOP의 특징에 관하여(상속성, 인터페이스)
***
   
## 인터페이스(Interface), 제어의 역전(Inversion of Control, IoC)
   
DOS에서의 프로그래밍은 절차지향으로 메인함수가 시작부터 끝까지 전체를 통제하는 스타일이었습니다. 
   
언제 입력을 받을 지, 언제 화면을 그릴 지 모두 해당됩니다. 
   
그래픽 유저 인터페이스(GUI), 이벤트 기반(Event-driven) 프로그래밍 쪽으로 넘어가면서 
플로우를 일직선으로 단순하게 표현하는 것이 불가능해집니다. 
언제 키보드 입력이 들어가는지, 마우스 입력이 들어가는지 같은 것들에 대한 예측이 불가능해지고 이를 
보완하기 위해서 무한루프를 돌리는 방식을 사용하게 됩니다. 
   
무한루프를 돌리면서 운영체제나 프레임워크에 어떤 입력이 들어올 경우 해당 입력에 대해서 
어떤 출력, 동작을 할 지 미리 프로그래밍을 해 두는 방식의 구현이 나타나게 됩니다. 
이는 미리 등록해둔 코드를 대신 호출해주는 방식으로 운영체제나 프레임워크가 시스템의 제어권을 
가지고 있다는 식으로 표현됩니다. 이것이 제어의 역전(Inversion of Control)입니다.
   
C언어에서의 제어의 역전은 함수 포인터(Function Pointer)를 활용, 콜백(Callback)구조의 구현을 통해서 
제어의 역전을 구현하였습니다.
   
예시는 아래와 같습니다. 
   
```C
#include <stdio.h>

// 시스템 제어권을 가지는 프레임워크 영역입니다. 
// 콜백 함수의 원형을 정의합니다. 클라이언트에 등록되는 내용입니다. 
// void* context를 사용하며 클라이언트의 데이터(상태)를 전달받을 수 있도록 구현합니다.
typedef void (*EventCallback)(void* context);

// 프레임워크가 보관할 함수 포인터, 데이터 포인터입니다.
EventCallback g_registeredCallback = NULL;
void* g_registeredContext = NULL;

// 콜백 등록 함수입니다. 의존성 주입에 해당합니다.
void RegisterEvent(EventCallback cb, void* context){
	g_registeredCallback = cb;
	g_registeredContext = context;
}

// 프레임워크 메인 루프입니다. 제어 흐름을 다룹니다.
void FrameworkRun(){
	printf("[Framework] 시스템 루프 대기 중...\n");

	// 이벤트 발생을 체크합니다.
	if (g_registeredCallback != NULL){
		printf("[Framework] 이벤트 감지, 제어권을 클라이언트로 넘깁니다.\n");

		// 보관해뒀던 void* 데이터를 인자로 넣습니다. 콜백을 실행합니다.
		g_registeredCallback(g_registerdContext);
	}
	printf("[Framework] 시스템 제어권 회수 및 종료\n");
}

// 클라쪽입니다. 사용자 작성의 데이터, 커스텀 로직입니다.
typedef struct{
	int hp;
	const char* name;
} PlayerData;

// 커스텀 콜백 함수입니다.
void MyCustomLogic(void* context){
	// void* 로 넘어온 데이터를 원래의 타입으로 캐스팅합니다.
	PlayerData* pData = (PlayerData*)context;
	pData->hp -= 10;
	printf("[Client] %s의 로직 실행, 남은 HP : %d\n", pData->name, pData->hp);
}

// 조립자. 메인 쪽입니다. 의존성 연결 후 실행합니다.

int main(){
	
	// 프레임워크에 클라 로직인 함수와 데이터(상태)를 주입합니다.
	PlayerData myPlayer = {100, "myCharacter"};

	// 이후 모든 실행 제어권을 프레임워크로 넘깁니다. 이렇게 제어의 역전이 이루어집니다.
	RegisterEvent(MyCustomLogic, &myPlayer);

	FrameworkRun();

	return 0;
}
```
   
다음은 C++에서의 제어의 역전 구현입니다.
   
```C++
#include <iostream>
#include <vector>
#include <memeory>

// 인터페이스 구현입니다. (IUpdatable.h)
class IUpdatable{

pulbic:
	virtual ~IUpdatable() = default;
	virtual void Update() = 0; // 엔진 호출 콜백 함수입니다.
};

// 실제 구현 로직입니다. (Player.h / .cpp) 상태와 행위를 가집니다.
class Player : public IUpdatable{

private:
	int hp = 100; // C언어 콜백과 다르게 객체 내부에 상태를 보관 가능합니다.
public:
	void Update() override{
		std::cout << "[Player] 입력을 받아 이동합니다. 현재 HP : " << hp << "\n";
	}
};

// 프레임 워크입니다. (GameEngine.h / .cpp) 제어권을 가진 주체에 해당합니다.
class GameEngine{

private:
	std::vector<std::shared_ptr<IUpdatable>> objects;
public:
	// 의존성 주입 파트입니다. 외부에서 객체를 밀어넣는다고 볼 수 있습니다.
	void RegisterObject(std::shared_ptr<IUpdatable> obj){
		object.push_back(obj);
	}

	// 엔진이 제어 흐름을 주도, 클라 코드를 대신 호출합니다.
	void Run(){
		std::cout << "게임 엔진 루프 시작 (제어권 역전)\n";
		for (const auto& obj : objects){
			obj->Update();
		}
	}
};

// 조립자 파트입니다. (main.cpp) 컴포넌트들을 연결합니다.
int main(){

	GameEngine engine;
	std::shared_ptr<IUpdatable> myPlayer = std::make_shared<Player>();
	
	// Player 객체의 제어권을 GameEngine으로 넘김으로써 제어의 역전을 발생시킵니다.
	engine.RegisterObject(myPlayer);

	// 사용자는 루프를 직접 돌리지 않으며 엔진을 켜기만 합니다.
	engine.Run();

	return 0;
}
```
   
제어의 역전으로 프레임 워크가 함수를 실행하는 것을 게임엔진에서 보면 다음과 같은 흐름을 가지게 됩니다.
   
>> 운영체제의 스레드 -> 게임 엔진의 메인 루프 -> 엔진의 이벤트 처리기 -> 구현해둔 콜백 함수(함수 포인터)
