# OOP의 특징에 관하여(정보 은닉)
***
   
## 정보 은닉(Data Hiding)
   
정보 은닉은 접근 제어 지시자(Access Modifiers)를 적용함으로 쓰임을 알 수 있습니다. 
외부에서 객체의 내부 데이터 조작 방지를 위해서 데이터에 해당하는 멤버 변수를 private으로 선언하는 
경우가 이에 해당합니다. 
   
내부 데이터를 숨기면서 객체는 외부 소통 창구로써 public method(Getter, Setter)를 이용케 합니다. 
해당 메서드들을 사용함으로써 내부 구조에 대해서 신경쓸 필요 없이 인터페이스만 알면 사용할 수 있도록 
기능을 제공합니다. 
   
프로젝트에서 정보 은닉을 통해 안정성을 얻게 합니다. 
상태 변경을 시도하려는 전체 접근을 특정 메서드로 모이게 하며(Choke point) 해당 포인트에서 
유효성 검증(플레이어 체력은 음수가 될 수 없다, 배열의 인덱스는 실제 크기를 초과할 수 없다 등..) 
을 진행하며 해당 검증을 통해서 객체의 무결성을 유지합니다. 
   
내부 데이터를 숨기게 될 경우 성능 향상을 위한 리펙토링을 한다고 할 때 자료구조를 완전히 엎는다고 해도 
파급효과(Ripple Effect)의 발생을 차단할 수 있습니다. 
std::vector에서 std::list의 사용으로 변경이 있더라도 외부에서는 동일하게 public 메서드를 호출하기만 하면 됩니다. 
   
>> 파급 효과(Ripple Effect)와 부수 효과(Side Effect)의 차이
   
- Ripple Effect
   
주요 관점 : 구조, 설계, 유지보수(개발 타임)
   
A라는 클래스나 모듈의 코드 변경 시 그와 의존성을 가진 B, C, D의 코드도 연달아 수정하고 
다시 컴파일해야하는 현상에 해당합니다. 정보 은닉(캡슐화), 결합도(Coupling)과 밀접한 관련이 있습니다. 
   
정보 은닉이 없어 리플 이펙트가 발생할 수 있는 예시는 다음과 같습니다. 
   
```C++
class Player{

public:
		// 내부 자료구조를 public으로 노출
	int inventory[10];
};

void useInventory(){
	Player p;
		// 외부 코드 수십, 수백 곳에서 이미 고정된 배열 방식과 크기(10)에 의존하여 작성
	for (int i = 0; i < 10; ++i){
		p.inventory[i] = 0;
	}
}
```
   
정보 은닉으로 리플 이펙트를 차단하는 예시는 다음과 같습니다. 
   
```C++
#include <iostream>

class Player{

private:
		// 내부 자료구조를 private으로 숨김
	std::vector<int> inventory;

pulbic:
	Player() : inventory(10, 0) {}

		// 외부와 소통하는 고정된 인터페이스만 제공
	int getItem(int idx) const{
		return inventory[idx];
	}

	void setItem(int idx, int value){
		inventory[idx] = value;
	}
};

void useInventory(){
	Player p;

		// 외부 코드는 내부가 배열인지 벡터인지 모른 채 인터페이스만 사용하는 구조, 이래도 됨
	p.setItem(0, 0);
	int item = p.getItem(0);
}
```
   
- Side Effect 
   
주요 관점 : 함수의 동작, 상태 변경(런타임 / 실행 타임)
   
함수 실행이 외부 상태를 변경하는 현상입니다. 
함수가 자신의 지역 변수(Local variable)나 반환값(return value) 외에 외부의 전역변수를 수정하거나, 
참조로 넘어온 인자의 값을 바꾸거나, 화면에 출력을 하는 등 외부 환경에 영향을 미치는 모든 행위를 사이드 이펙트로 묶어서 봅니다. 
주로 순수 함수(Pure Function)나 함수형 프로그래밍에서 중요히 다룹니다. 
   
사이드 이펙트의 발생 예시는 다음과 같습니다. 
   
```C++
#include <iostream>

int global_count = 0;	// 외부 상태 (전역 변수)

	// 사이드 이펙트가 없는 순수 함수(Pure Function)
	// 입력값만으로 결과가 결정되며, 외부의 어떤 것도 건드리지 않습니다. 
int add(int a, int b){
	return a + b;
}

	// 사이드 이펙드가 있는 함수
	// 덧셈 결과를 반환하는 것 외에 외부 상태를 변경하는 부수적인 작용을 합니다.
int addWithSideEffect(int a, int b){
	// 사이드 이펙트 1. 함수 외부의 전역 변수 값을 임의로 변경하였습니다.
	global_count++;
	
	// 사이드 이펙트 2. 콘솔(외부 출력 장치)의 상태를 변경하였습니다.
	std::cout << "덧셈이 수행되었습니다. \n";

	return a + b;
}

int main(){
		// 사이드 이펙트가 발생하지 않는 순수 함수 호출입니다.
	int result1 = add(3, 4);

		// 사이드 이펙트 발생 함수 호출입니다.
		// 결과값 7을 반환하나 동시에 global_count가 1 증가하고 화면에 문자가 출력됩니다._
	int result2 = addWithSideEffect(3, 4);

	std::cout << "현재 호출 횟수 : " << global_count << "\n";

	return 0;
}
```
   
사이드 이펙트는 나쁜 현상이 아닙니다. 
화면에 글자를 출력하거나, 파일에 데이터를 쓰거나 데이터베이스를 업데이트 하는 행위 모두가 
사이드 이펙트에 해당합니다. 프로그램이 유의미한 동작을 하기 위해서는 필수적입니다.
다만 의도하지 않은 사이드 이펙트를 통제하고 예측 가능한 프로그래밍을 하는 것이 유의됩니다. 
위의 addWithSideEffect 함수의 동작을 숫자 두 개를 더하는 용도로만 알고 있는 상태로 
이 함수를 반복문 안에서 수십 수천 호출하게 된다면 의도치 않게 global_count값이 너무 큰 값으로 
변화하게 되며 콘솔 출력이 도배되게 되는데 이는 예측 불가능하기에 좋지 않은 결과로 귀결됩니다.

C++ 프로젝트에서 헤더 파일에 private 멤버 변수의 존재도 숨기려고 실제 구현부를 다른 클래스로 빼고 
포인터만 들고 있는 Pimpl(Pointer to Implementation)로 구성하는 방식이 자주 사용됩니다. 
이를 통해 컴파일 의존성을 낮추며 빌드 시간을 단축시킬 수 있습니다. 
   
정보 은닉이 구현된 다른 예시힙니다. 
   
```C++
#include <iostream>
#include <string>
#include <algorithm>

class Character{

private:
		// 외부에서 직접 접근 불가능한 데이터
	std::string name;
	int health;
	int maxHealth;

		// 내부에서만 사용하는 유틸리티 함수(세부 구현 은닉)
	void clampHealth(){
		health = std::max(0, std::min(health, maxHealth));
	}

public:
	Character(std::string charName, int initialMaxHealth)
		: name(charName), maxHealth(initialMaxHealth), health(initialMaxHealth) {}

		// 상태를 읽기만 가능하게 하는 읽기 전용 인터페이스
	std::string getName() const { return name; }
	int getHealth const { return health }
		
		// 외부 소통 행위(Behavior) 메서드
		// 비즈니스 로직, 안전 장치가 포함됨. 단순 대입x
	void takeDamage(int damage){
		if (damage <= 0) return;	// 잘못된 입력 차단

		std::cout << name "이(가)" << damage << "의 피해를 입었습니다. \n";
		health -= damage;
		clampHealth():	// 내부 상태를 안전한 범위로 보정(불변성 유지)

		if (health == 0){
			std:cout << name << "이(가) 쓰러졌습니다. \n";
		}
 	}

	void heal(int amount){
		if (amount <= 0) return;
		
		std::cout << name "이(가)" << amount << "의 체력을 회복했습니다. \n";
		health += amount;
		clampHealth();
	}
};

int main(){
		// Character 객체는 오직 주어진 메서드를 통해서만 상호작용
	Character myCharacter("내 캐릭터", 100);
	
	myCharacter.takeDamage(30);
	std::cout << "현재 체력 : " << myCharacter.getHealth() << "\n";

		// myCharacter.health = 10000; // 이 경우 컴파일 에러가 발생합니다. 내부 데이터를 임의로 조작하는 것이 불가능하기 때문

	myCharacter.heal(40);
	std::cout << "현재 체력 : " << myCharacter.getHealth() << "\n";

	return 0;
```
