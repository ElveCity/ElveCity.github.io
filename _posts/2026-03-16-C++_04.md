# OOP의 특징에 관하여(캡슐화)
***
   
## 캡슐화(Encapsulation)
   
C언어에서는 전역 변수나 구조체 형태 데이터와 데이터 조작을 위한 전역 함수들이 개별로 나눠져 있었습니다. 
프로젝트의 규모가 커짐에 따라서 데이터와 그 데이터 조작 함수간의 연관성을 추적하기 힘들어지는 응집도 문제가 발생합니다. 
   
C++ 기준으로 캡슐화는 클래스를 기반으로 구현됩니다. 
상태를 나타내는 멤버 변수(Data Members)와 그 상태의 조작을 위한 멤버 함수(Member Functions, Methods) 
를 하나의 클래스로 묶는 것을 캡슐 안에 가둔다고 캡슐화라고 합니다. 
   
다음의 코드는 캡슐화가 되지 않은 상태의 C언어 코드입니다.
```C
#incldue <stdio.h>

	// 캡슐화 기준 public 상태에 해당합니다. 내부 상태의 보호가 불가능합니다.
typedef struct{
	char name[20];
	int hp;
} Player;

	// 외부 함수에 해당합니다. 객체 외부에서 포인터를 통해 메모리를 직접 덮어 쓰는 경우에 해당합니다. 
void TakeDamage(Player* p, int damage){
	p->hp -= damage;	// 
}

int main(){
	Player p1 = {"Character01", 100};

	TakeDamage(&p1, 30);	// 정상 외부 함수 호출
	printf("남은 체력 : &d\n", p1.hp); // p1.hp = 70

	p1.hp = -1000; // 외부에서 메모리 주소에 직접 접근, 임의 조작

	printf("체력 값이 비정상적입니다. HP : %d\n", p1.hp); // p1.hp = -1000, 외부 조작값이 그대로 적용된 상태

	return 0;
}
```

다음의 코드는 캡슐화가 이루어진 상태의 C++ 코드입니다.
   
```C++
#include <iostream>
#include <string>
#include <algorithm>

class Player{

private:
	std::string name;
	int hp;

public:
	Player(std:: pName, int initialHp) : name(pName), hp(initialHp) {}

	void TakeDamage(int damage){
		if (damage < 0) return;		// 잘못된 값이 들어가는 것을 방지, 유효성 검증에 해당

		hp -= damage;
		hp = std::max(0, hp);	// HP가 0 미만으로 떨어지지 않도록 무결성 유지

		std::cout << name << "데미지를 입었습니다. 현재 체력은" << hp << "입니다\n";
	}

	int GepHp() const { return hp; }	// 외부에는 오직 읽기 권한(const Getter)만 제한적으로 제공
};

int main(){
	Player p1("Character01", 100);

	p1.TakeDamage(30);	// private 멤버에 접근, 컴파일 에러

	std::cout << "남은 체력은" << p1.GetHp() << "입니다.\n";

	return 0;
}
```
